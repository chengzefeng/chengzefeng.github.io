常用的Math对象方法

方法 描述 备注
abs(x) 返回数的绝对值
ceil(x) 对数进行上舍入 console.log(Math.ceil(3.1)) 结果：4
floor(x) 对数进行下舍入 同ceil方法相反
max(a,b) 取a,b中的最大值
min(a,b) 取a,b中的最小值
random() 返回0-1之间的随机数
round(x) 四舍五入

DOM

属性/方法 描述 备注
element.appendChild() 向元素添加新的子节点，作为最后一个子节点
element.className 设置或返回元素的 class 属性
element.clientHeight 返回元素的可见高度
element.firstChild 返回指定节点的首个子节点
element.getAttribute() 返回元素节点的指定属性值
element.offsetHeight 返回元素的高度
element.offsetWidth 返回元素的宽度
element.offsetLeft 返回元素的水平偏移位置
element.offsetTop 返回元素的垂直偏移位置
element.removeChild() 从元素中移除子节点

放大镜

<!DOCTYPE html>
<html>
    <head>
        <meta charset="utf-8"/>
        <title>放大镜</title>
        <style type="text/css">
            *{margin:0px;padding: 0px;}
            .small{width: 179px;height: 214px;overflow: hidden;cursor: move;position: relative;}
            .pointer{width: 50px;height: 50px;background: rgba(0,0,255,0.3);position: absolute;top: 0;left: 0;display: none;}
            img{border:none;border: 0;}
            .small img{width: 179px;height: 214px;float: left;}
            .big{width: 179px;height: 214px;border: 1px solid black;position: absolute;left: 200px;top:0;display: none;position:absolute;overflow: hidden;}
            .big img{position: absolute;left: 0;top:0;}
        </style>
    </head>
    <body>
        <div class="fangda">
            <div class="small">
                <img src="2.jpg" alt="">
                <span class="pointer"></span>
            </div>
            <div class="big">
                <img src="图片路径" alt="">
            </div>
        </div>
        <script type="text/javascript">
            window.onload=function(){
                var small=document.querySelector('.small');
                var smallImg=document.querySelector('.small img');
                var bigImg=document.querySelector('.big img');
                var pointer=document.querySelector('.pointer');
                var big=document.querySelector('.big');
                small.onmouseover=function(){
                    pointer.style.display='block';
                    big.style.display='block';
                }
                small.onmouseout=function(){
                    pointer.style.display='none';
                    big.style.display='none';
                }
                small.onmousemove=function(event){

                    var oEvent= event || window.event;
                    var x=oEvent.clientX-pointer.offsetWidth/2;
                    var y=oEvent.clientY-pointer.offsetWidth/2;

                    var x1=oEvent.clientX;
                    var y1=oEvent.clientY;

                    pointer.innerText="x1:"+x1+"y1:"+y1+" "+"x:"+x+"y:"+y;



                    pointer.style.left=x+"px";
                    pointer.style.top=y+"px";


                    if(x<=0){
                        pointer.style.left=0+"px";
                    }else if(x>=small.offsetWidth-pointer.offsetWidth){
                        pointer.style.left=(small.offsetWidth-pointer.offsetWidth)+"px";
                    }
                    if(y<=0){
                        pointer.style.top=0+"px";
                    }else if(y>=small.offsetHeight-pointer.offsetHeight){
                        pointer.style.top=(small.offsetHeight-pointer.offsetHeight)+"px";
                    }


                    // 竖直方向放大倍数
                    var l=small.offsetWidth/pointer.offsetWidth;
                    // 水平方向放大倍数
                    var s=small.offsetHeight/pointer.offsetHeight;

                    bigImg.style.width=l*smallImg.offsetWidth+"px";
                    bigImg.style.height=s*smallImg.offsetHeight+"px";


                    bigImg.style.top=-s*pointer.offsetTop+"px";
                    bigImg.style.left=-l*pointer.offsetLeft+"px"
                }
            }
        </script>
    </body>
</html>
